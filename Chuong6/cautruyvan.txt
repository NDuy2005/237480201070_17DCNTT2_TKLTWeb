Câu 1: tạo thêm bảng tồn kho
CREATE TABLE TONKHO (
    mahang VARCHAR(4) PRIMARY KEY,
    tenhang VARCHAR(40),
    mansx VARCHAR(2),
    namsx INT,
    gia INT,
    soluong INT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

Câu 4: Xóa khách hàng có mã K010 trong KHACHHANG
-- Lưu ý: Nếu mã K010 đã có trong bảng HOADON, 
-- bạn cần xóa dữ liệu ở bảng HOADON trước do ràng buộc khóa ngoại.
DELETE FROM KHACHHANG 
WHERE makh = 'K010';

Câu 5: Hiển thị những hàng hóa có nhà sản xuất là "DELL"
SELECT HANGHOA.* FROM HANGHOA 
INNER JOIN NHASANXUAT ON HANGHOA.mansx = NHASANXUAT.mansx 
WHERE NHASANXUAT.tennsx = 'DELL';
